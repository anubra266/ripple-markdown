## Install

```bash
npm i --save ripple-markdown
#or
yarn add ripple-markdown
#or
pnpm add ripple-markdown
```

## Usage

```jsx
import RippleMarkdown from "ripple-markdown";

export component Page() {
  <RippleMarkdown markdown={`# My markdown Heading`} />
}
```
